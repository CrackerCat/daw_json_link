169  template<typename Policy, typename... Policies>
170  struct policy_bits_start_impl<Policy, std::tuple<Policies...>> {
171  static constexpr auto idx =
172  traits::pack_index_of_v<Policy, Policies...>;
173  static_assert( idx >= 0, "Policy is not registered" );
174  using tp_policies = std::tuple<Policies...>;
175 
176  template<std::size_t Pos, int End>
177  static constexpr unsigned do_step( ) {
178  if constexpr( Pos >= static_cast<std::size_t>( End ) ) {
179  return 0U;
180  }
181  return policy_bits_width<std::tuple_element_t<Pos, tp_policies>>;
182  }
183 
184  template<std::size_t... Is>
185  static constexpr unsigned calc( std::index_sequence<Is...> ) {
186  return ( do_step<Is, idx>( ) + ... );
187  }
188  };

219  template<typename Policy>
220  constexpr void set_bits_in( policy_options_t &value, Policy e ) {
221  static_assert( is_policy_flag<Policy>,
222  "Only registered policy types are allowed" );
223  unsigned new_bits = static_cast<unsigned>( e );
224  constexpr unsigned mask = (1U << policy_bits_width<Policy>)-1U;
225  new_bits &= mask;
226  new_bits <<= policy_bits_start<Policy>;
227  value &= ~mask;
228  value |= new_bits;
229  }

231  template<typename Policy, typename... Policies>
232  constexpr policy_options_t set_bits( policy_options_t value, Policy pol,
233  Policies... pols ) {
234  static_assert( ( is_policy_flag<Policies> and ... ),
235  "Only registered policy types are allowed" );
236 
237  unsigned new_bits = static_cast<unsigned>( pol );
238  constexpr unsigned mask = (1U << policy_bits_width<Policy>)-1U;
239  new_bits &= mask;
240  new_bits <<= policy_bits_start<Policy>;
241  value &= ~mask;
242  value |= new_bits;
243  if constexpr( sizeof...( Policies ) > 0 ) {
244  if constexpr( sizeof...( pols ) > 0 ) {
245  return set_bits( value, pols... );

279  constexpr Result get_bits_for( policy_options_t value ) {
280  static_assert( is_policy_flag<Policy>,

294  /***
295  * Create the parser options flag for BasicParsePolicy
296  * @tparam Policies Policy types that satisfy the `is_policy_flag` trait.
297  * @param policies A list of parser options to change from the defaults.
298  * @return A policy_options_t that encodes the options for the parser
299  */
300  template<typename... Policies>
301  constexpr json_details::policy_options_t
302  parse_options( Policies... policies ) {
303  static_assert( ( json_details::is_policy_flag<Policies> and ... ),
304  "Only registered policy types are allowed" );
305  auto result = json_details::default_policy_flag;
306  if constexpr( sizeof...( Policies ) > 0 ) {
307  result |= ( json_details::set_bits_for( policies ) | ... );
308  }
309  return result;
