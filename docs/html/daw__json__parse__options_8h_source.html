216  template<typename Policy, typename... Policies>
217  struct policy_bits_start_impl<Policy, std::tuple<Policies...>> {
218  static constexpr auto idx =
219  traits::pack_index_of_v<Policy, Policies...>;
220  static_assert( idx >= 0, "Policy is not registered" );
221  using tp_policies = std::tuple<Policies...>;
222 
223  template<std::size_t Pos, int End>
224  static constexpr unsigned do_step( ) {
225  if constexpr( Pos >= static_cast<std::size_t>( End ) ) {
226  return 0U;
227  }
228  return policy_bits_width<std::tuple_element_t<Pos, tp_policies>>;
229  }
230 
231  template<std::size_t... Is>
232  static constexpr unsigned calc( std::index_sequence<Is...> ) {
233  return ( do_step<Is, idx>( ) + ... );
234  }
235  };

278  template<typename Policy>
279  constexpr void set_bits_in( policy_options_t &value, Policy e ) {
280  static_assert( is_policy_flag<Policy>,
281  "Only registered policy types are allowed" );
282  unsigned new_bits = static_cast<unsigned>( e );
283  constexpr unsigned mask = (1U << policy_bits_width<Policy>)-1U;
284  new_bits &= mask;
285  new_bits <<= policy_bits_start<Policy>;
286  value &= ~mask;
287  value |= new_bits;
288  }

290  template<typename Policy, typename... Policies>
291  constexpr policy_options_t set_bits( policy_options_t value, Policy pol,
292  Policies... pols ) {
293  static_assert( ( is_policy_flag<Policies> and ... ),
294  "Only registered policy types are allowed" );
295 
296  unsigned new_bits = static_cast<unsigned>( pol );
297  constexpr unsigned mask = ( (1U << policy_bits_width<Policy>)-1U )
298  << policy_bits_start<Policy>;
299  new_bits &= mask;
300  new_bits <<= policy_bits_start<Policy>;
301  value &= ~mask;
302  value |= new_bits;
303  if constexpr( sizeof...( Policies ) > 0 ) {
304  if constexpr( sizeof...( pols ) > 0 ) {
305  return set_bits( value, pols... );
306  } else {
307  return value;
308  }
309  } else {
310  return value;
311  }
312  }

354  /***
355  * Create the parser options flag for BasicParsePolicy
356  * @tparam Policies Policy types that satisfy the `is_policy_flag` trait.
357  * @param policies A list of parser options to change from the defaults.
358  * @return A policy_options_t that encodes the options for the parser
359  */
360  template<typename... Policies>
361  constexpr json_details::policy_options_t
362  parse_options( Policies... policies ) {
363  static_assert( ( json_details::is_policy_flag<Policies> and ... ),
364  "Only registered policy types are allowed" );
365  auto result = json_details::default_policy_flag;
366  if constexpr( sizeof...( Policies ) > 0 ) {
367  result |= ( json_details::set_bits_for( policies ) | ... );
368  }
369  return result;
