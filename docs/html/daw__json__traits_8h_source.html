81  /***
82  * Default Constructor for a type. It accounts for aggregate types and uses
83  * brace construction for them
84  * @tparam T type to construct
85  */
86  template<typename T>
87  struct default_constructor {
88  template<typename... Args,
89  std::enable_if_t<std::is_constructible<T, Args...>::value,
90  std::nullptr_t> = nullptr>
91  [[nodiscard]] DAW_ATTRIBUTE_FLATTEN inline constexpr T
92  operator( )( Args &&...args ) const {
93 
94  return T( DAW_FWD( args )... );
95  }
96 
97  template<typename... Args,
98  typename std::enable_if_t<
99  std::conjunction<
100  daw::not_trait<std::is_constructible<T, Args...>>,
101  daw::traits::is_list_constructible<T, Args...>>::value,
102  std::nullptr_t> = nullptr>
103  [[nodiscard]] DAW_ATTRIBUTE_FLATTEN inline constexpr T
104  operator( )( Args &&...args ) const
105  noexcept( noexcept( T{ DAW_FWD( args )... } ) ) {
106 
107  return T{ DAW_FWD( args )... };
108  }
109  };
