139  DAW_ATTRIB_FLATINLINE inline std::vector<T, Alloc> operator( )( ) const
140  noexcept( noexcept( std::vector<T, Alloc>( ) ) ) {

144  DAW_ATTRIB_FLATINLINE inline std::vector<T, Alloc> &&
145  operator( )( std::vector<T, Alloc> &&v ) const
146  noexcept( noexcept( std::vector<T, Alloc>( v ) ) ) {
147  return DAW_MOVE( v );
148  }
149 
150  template<typename Iterator>
151  DAW_ATTRIB_FLATINLINE inline std::vector<T, Alloc>
152  operator( )( Iterator first, Iterator last,
153  Alloc const &alloc = Alloc{ } ) const
154  noexcept( noexcept( std::vector<T, Alloc>( first, last, alloc ) ) ) {
155  if constexpr( std::is_same_v<std::random_access_iterator_tag,
156  typename std::iterator_traits<
157  Iterator>::iterator_category> or
158  not json_details::is_std_allocator<Alloc>::value ) {
159  return std::vector<T, Alloc>( first, last, alloc );
160  } else {
161  constexpr auto reserve_amount = 4096U / ( sizeof( T ) * 8U );
162  auto result = std::vector<T, Alloc>( alloc );
163  // Lets use a WAG and go for a 4k page size
164  result.reserve( reserve_amount );
165  result.assign( first, last );
166  return result;
167  }
168  }
