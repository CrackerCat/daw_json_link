27  /***
28  * Handles the bounds and policy items for parsing execution and comments.
29  * @tparam IsUncheckedInput If true, do not perform all validity checks on
30  * input. This implies that we can trust the source to be perfect
31  * @tparam CommentPolicy The policy that handles skipping whitespace where
32  * comments may or may not be allowed:w
33  * @tparam ExecMode A Policy type for selecting if we must be constexpr, can
34  * use C/C++ runtime only methods, or if SIMD intrinsics are allowed
35  * @tparam AllowEscapedNames Are escapes allowed in member names. When true,
36  * the slower string parser is used
37  */

85  template<std::size_t N>
86  constexpr bool starts_with( char const ( &rhs )[N] ) const {
87  if( size( ) < ( N - 1 ) ) {
88  return false;
89  }
90  bool result = true;
91  for( std::size_t n = 0; n < ( N - 1 ); ++n ) {
92  result = result & ( first[n] == rhs[n] );
93  }
94  return result;
