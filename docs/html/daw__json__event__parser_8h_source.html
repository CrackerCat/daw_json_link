359  template<typename ParsePolicy = NoCommentSkippingPolicyChecked,
360  typename StackContainerPolicy = DefaultJsonEventParserStackPolicy<
361  JsonEventParserStackValue<ParsePolicy>>,
362  typename Handler>
363  inline constexpr void
364  json_event_parser( basic_json_value<ParsePolicy> jvalue,
365  Handler &&handler ) {
366  using iterator = basic_json_value_iterator<ParsePolicy>;
367  using json_value_t = basic_json_pair<ParsePolicy>;
368  using stack_value_t = JsonEventParserStackValue<ParsePolicy>;
369 
370  auto parent_stack = StackContainerPolicy( );
371 
372  auto const move_to_last = [&]( ) {
373  parent_stack.back( ).value.first = parent_stack.back( ).value.second;
374  };
375 
376  auto const process_value = [&]( json_value_t p ) {
377  {
378  auto result = json_details::handle_on_value( handler, p );
379  switch( result.value ) {
380  case json_parse_handler_result::Complete:
381  parent_stack.clear( );
382  return;
383  case json_parse_handler_result::SkipClassArray:
384  move_to_last( );
385  return;
386  case json_parse_handler_result::Continue:
387  break;
388  }
389  }
390 
391  auto &jv = p.value;
392  switch( jv.type( ) ) {
393  case JsonBaseParseTypes::Array: {
394  auto result = json_details::handle_on_array_start( handler, jv );
395  switch( result.value ) {
396  case json_parse_handler_result::Complete:
397  parent_stack.clear( );
398  return;
399  case json_parse_handler_result::SkipClassArray:
400  move_to_last( );
401  return;
402  case json_parse_handler_result::Continue:
403  break;
404  }
405  parent_stack.push_back(
406  { StackParseStateType::Array,
407  std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
408  } break;
409  case JsonBaseParseTypes::Class: {
410  auto result = json_details::handle_on_class_start( handler, jv );
411  switch( result.value ) {
412  case json_parse_handler_result::Complete:
413  parent_stack.clear( );
414  return;
415  case json_parse_handler_result::SkipClassArray:
416  move_to_last( );
417  return;
418  case json_parse_handler_result::Continue:
419  break;
420  }
421  parent_stack.push_back(
422  { StackParseStateType::Class,
423  std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
424  } break;
425  case JsonBaseParseTypes::Number: {
426  auto result = json_details::handle_on_number( handler, jv );
427  switch( result.value ) {
428  case json_parse_handler_result::Complete:
429  parent_stack.clear( );
430  return;
431  case json_parse_handler_result::SkipClassArray:
432  move_to_last( );
433  return;
434  case json_parse_handler_result::Continue:
435  break;
436  }
437  } break;
438  case JsonBaseParseTypes::Bool: {
439  auto result = json_details::handle_on_bool( handler, jv );
440  switch( result.value ) {
441  case json_parse_handler_result::Complete:
442  parent_stack.clear( );
443  return;
444  case json_parse_handler_result::SkipClassArray:
445  move_to_last( );
446  return;
447  case json_parse_handler_result::Continue:
448  break;
449  }
450  } break;
451  case JsonBaseParseTypes::String: {
452  auto result = json_details::handle_on_string( handler, jv );
453  switch( result.value ) {
454  case json_parse_handler_result::Complete:
455  parent_stack.clear( );
456  return;
457  case json_parse_handler_result::SkipClassArray:
458  move_to_last( );
459  return;
460  case json_parse_handler_result::Continue:
461  break;
462  }
463  } break;
464  case JsonBaseParseTypes::Null: {
465  auto result = json_details::handle_on_null( handler, jv );
466  switch( result.value ) {
467  case json_parse_handler_result::Complete:
468  parent_stack.clear( );
469  return;
470  case json_parse_handler_result::SkipClassArray:
471  move_to_last( );
472  return;
473  case json_parse_handler_result::Continue:
474  break;
475  }
476  } break;
477  case JsonBaseParseTypes::None:
478  default: {
479  auto result = json_details::handle_on_error( handler, jv );
480  switch( result.value ) {
481  case json_parse_handler_result::Complete:
482  parent_stack.clear( );
483  return;
484  case json_parse_handler_result::SkipClassArray:
485  move_to_last( );
486  return;
487  case json_parse_handler_result::Continue:
488  break;
489  }
490  } break;
491  }
492  };
493 
494  auto const process_range = [&]( stack_value_t v ) {
495  if( v.value.first != v.value.second ) {
496  auto jv = *v.value.first;
497  v.value.first++;
498  parent_stack.push_back( DAW_MOVE( v ) );
499  process_value( DAW_MOVE( jv ) );
500  } else {
501  switch( v.type ) {
502  case StackParseStateType::Class: {
503  auto result = json_details::handle_on_class_end( handler );
504  switch( result.value ) {
505  case json_parse_handler_result::Complete:
506  parent_stack.clear( );
507  return;
508  case json_parse_handler_result::SkipClassArray:
509  case json_parse_handler_result::Continue:
510  break;
511  }
512  } break;
513  case StackParseStateType::Array: {
514  auto result = json_details::handle_on_array_end( handler );
515  switch( result.value ) {
516  case json_parse_handler_result::Complete:
517  parent_stack.clear( );
518  return;
519  case json_parse_handler_result::SkipClassArray:
520  case json_parse_handler_result::Continue:
521  break;
522  }
523  } break;
524  }
525  }
526  };
527 
528  process_value( { std::nullopt, DAW_MOVE( jvalue ) } );
529 
530  while( not parent_stack.empty( ) ) {
531  auto v = DAW_MOVE( parent_stack.back( ) );
532  parent_stack.pop_back( );
533  process_range( v );
534  }
535  }
