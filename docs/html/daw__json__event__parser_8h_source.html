348  template<typename ParsePolicy = NoCommentSkippingPolicyChecked,
349  typename StackContainerPolicy = DefaultJsonEventParserStackPolicy<
350  JsonEventParserStackValue<ParsePolicy>>,
351  typename Handler>
352  inline constexpr void
353  json_event_parser( DAW_JSON_NS::basic_json_value<ParsePolicy> jvalue,
354  Handler &&handler ) {
355  using iterator = DAW_JSON_NS::basic_json_value_iterator<ParsePolicy>;
356  using json_value_t = DAW_JSON_NS::basic_json_pair<ParsePolicy>;
357  using stack_value_t = JsonEventParserStackValue<ParsePolicy>;
358 
359  auto parent_stack = StackContainerPolicy( );
360 
361  auto const move_to_last = [&]( ) {
362  parent_stack.back( ).value.first = parent_stack.back( ).value.second;
363  };
364 
365  auto const process_value = [&]( json_value_t p ) {
366  {
367  auto result = json_details::handle_on_value( handler, p );
368  switch( result.value ) {
369  case json_parse_handler_result::Complete:
370  parent_stack.clear( );
371  return;
372  case json_parse_handler_result::SkipClassArray:
373  move_to_last( );
374  return;
375  case json_parse_handler_result::Continue:
376  break;
377  }
378  }
379 
380  auto &jv = p.value;
381  switch( jv.type( ) ) {
382  case DAW_JSON_NS::JsonBaseParseTypes::Array: {
383  auto result = json_details::handle_on_array_start( handler, jv );
384  switch( result.value ) {
385  case json_parse_handler_result::Complete:
386  parent_stack.clear( );
387  return;
388  case json_parse_handler_result::SkipClassArray:
389  move_to_last( );
390  return;
391  case json_parse_handler_result::Continue:
392  break;
393  }
394  parent_stack.push_back(
395  { StackRangeType::Array,
396  std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
397  } break;
398  case DAW_JSON_NS::JsonBaseParseTypes::Class: {
399  auto result = json_details::handle_on_class_start( handler, jv );
400  switch( result.value ) {
401  case json_parse_handler_result::Complete:
402  parent_stack.clear( );
403  return;
404  case json_parse_handler_result::SkipClassArray:
405  move_to_last( );
406  return;
407  case json_parse_handler_result::Continue:
408  break;
409  }
410  parent_stack.push_back(
411  { StackRangeType::Class,
412  std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
413  } break;
414  case DAW_JSON_NS::JsonBaseParseTypes::Number: {
415  auto result = json_details::handle_on_number( handler, jv );
416  switch( result.value ) {
417  case json_parse_handler_result::Complete:
418  parent_stack.clear( );
419  return;
420  case json_parse_handler_result::SkipClassArray:
421  move_to_last( );
422  return;
423  case json_parse_handler_result::Continue:
424  break;
425  }
426  } break;
427  case DAW_JSON_NS::JsonBaseParseTypes::Bool: {
428  auto result = json_details::handle_on_bool( handler, jv );
429  switch( result.value ) {
430  case json_parse_handler_result::Complete:
431  parent_stack.clear( );
432  return;
433  case json_parse_handler_result::SkipClassArray:
434  move_to_last( );
435  return;
436  case json_parse_handler_result::Continue:
437  break;
438  }
439  } break;
440  case DAW_JSON_NS::JsonBaseParseTypes::String: {
441  auto result = json_details::handle_on_string( handler, jv );
442  switch( result.value ) {
443  case json_parse_handler_result::Complete:
444  parent_stack.clear( );
445  return;
446  case json_parse_handler_result::SkipClassArray:
447  move_to_last( );
448  return;
449  case json_parse_handler_result::Continue:
450  break;
451  }
452  } break;
453  case DAW_JSON_NS::JsonBaseParseTypes::Null: {
454  auto result = json_details::handle_on_null( handler, jv );
455  switch( result.value ) {
456  case json_parse_handler_result::Complete:
457  parent_stack.clear( );
458  return;
459  case json_parse_handler_result::SkipClassArray:
460  move_to_last( );
461  return;
462  case json_parse_handler_result::Continue:
463  break;
464  }
465  } break;
466  case DAW_JSON_NS::JsonBaseParseTypes::None:
467  default: {
468  auto result = json_details::handle_on_error( handler, jv );
469  switch( result.value ) {
470  case json_parse_handler_result::Complete:
471  parent_stack.clear( );
472  return;
473  case json_parse_handler_result::SkipClassArray:
474  move_to_last( );
475  return;
476  case json_parse_handler_result::Continue:
477  break;
478  }
479  } break;
480  }
481  };
482 
483  auto const process_range = [&]( stack_value_t v ) {
484  if( v.value.first != v.value.second ) {
485  auto jv = *v.value.first;
486  v.value.first++;
487  parent_stack.push_back( DAW_MOVE( v ) );
488  process_value( DAW_MOVE( jv ) );
489  } else {
490  switch( v.type ) {
491  case StackRangeType::Class: {
492  auto result = json_details::handle_on_class_end( handler );
493  switch( result.value ) {
494  case json_parse_handler_result::Complete:
495  parent_stack.clear( );
496  return;
497  case json_parse_handler_result::SkipClassArray:
498  case json_parse_handler_result::Continue:
499  break;
500  }
501  } break;
502  case StackRangeType::Array: {
503  auto result = json_details::handle_on_array_end( handler );
504  switch( result.value ) {
505  case json_parse_handler_result::Complete:
506  parent_stack.clear( );
507  return;
508  case json_parse_handler_result::SkipClassArray:
509  case json_parse_handler_result::Continue:
510  break;
511  }
512  } break;
513  }
514  }
515  };
516 
517  process_value( { std::nullopt, DAW_MOVE( jvalue ) } );
518 
519  while( not parent_stack.empty( ) ) {
520  auto v = DAW_MOVE( parent_stack.back( ) );
521  parent_stack.pop_back( );
522  process_range( v );
523  }
524  }
