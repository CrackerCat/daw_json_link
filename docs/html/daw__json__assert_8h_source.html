147  Range const &location ) {
148 #ifdef DAW_USE_JSON_EXCEPTIONS
149  if constexpr( ShouldThrow ) {
150  if( location.first ) {

154  auto loc = std::string_view( location.class_first, len );
155  throw daw::json::json_exception( reason, loc );
