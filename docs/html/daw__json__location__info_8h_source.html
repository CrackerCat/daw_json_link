80  template<std::size_t start_pos>

82  find_name( daw::string_view key ) const {
83  UInt32 const hash = name_hash( key );
84 #if defined( _MSC_VER ) and not defined( __clang__ )
85  // MSVC has a bug where the list initialization isn't sequenced in order
86  // of appearance.
87  for( std::size_t n = 0; n < MemberCount; ++n ) {
88 #else
89  for( std::size_t n = start_pos; n < MemberCount; ++n ) {
90 #endif
91  if( hashes[n] == hash ) {
92  if constexpr( has_collisions ) {
93  if( DAW_JSON_UNLIKELY( key != names[n].name ) ) {
94  continue;
