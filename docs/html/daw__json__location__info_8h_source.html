81  template<std::size_t start_pos>
82  [[nodiscard]] constexpr std::size_t
83  find_name( daw::string_view key ) const {
84  UInt32 const hash = name_hash( key );
85 #if defined( _MSC_VER ) and not defined( __clang__ )
86  // MSVC has a bug where the list initialization isn't sequenced in order
87  // of appearance.
88  for( std::size_t n = 0; n < MemberCount; ++n ) {
89 #else
90  for( std::size_t n = start_pos; n < MemberCount; ++n ) {
91 #endif
92  if( hashes[n] == hash ) {
93  if constexpr( has_collisions ) {
94  if( DAW_JSON_UNLIKELY( key != names[n].name ) ) {
95  continue;
96  }
97  }
98  return n;
99  }
100  }
101  return MemberCount;
102  }

142  bool is_nullable, daw::string_view member_name ) {
143 
144  daw_json_assert_weak( is_nullable or ( not locations[pos].missing( ) ) or
145  ( not rng.is_closing_brace_checked( ) ),
146  missing_member( member_name ), rng );

148  rng.trim_left_unchecked( );
149  // TODO: should we check for end
150  while( locations[pos].missing( ) & ( rng.front( ) != '}' ) ) {
151  daw_json_assert_weak( rng.has_more( ), ErrorReason::UnexpectedEndOfData,
152  rng );
153  // TODO: fully unescape name
154  auto const name = parse_name( rng );
155  auto const name_pos =
156  locations.template find_name<( from_start ? 0 : pos )>( name );
157  if( name_pos >= std::size( locations ) ) {
158  // This is not a member we are concerned with
159  (void)skip_value( rng );
160  rng.clean_tail( );
161  continue;
162  }
163  if( name_pos == pos ) {
164  locations[pos].template set_range( rng );
165  break;
166  } else {
167  // We are out of order, store position for later
168  // OLDTODO: use type knowledge to speed up skip
169  // OLDTODO: on skipped classes see if way to store

172  // RESULT: storing preparsed is slower, don't try 3 times
173  // it also limits the type of things we can parse potentially

175  locations[name_pos].set_range( skip_value( rng ) );
