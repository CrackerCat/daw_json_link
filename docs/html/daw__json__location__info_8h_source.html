110  template<std::size_t start_pos>
111  [[nodiscard]] constexpr std::size_t
112  find_name( daw::string_view key ) const {
113  UInt32 const hash = name_hash( key );
114 #if defined( _MSC_VER ) and not defined( __clang__ )
115  // MSVC has a bug where the list initialization isn't sequenced in
116  // order of appearance.
117  for( std::size_t n = 0; n < MemberCount; ++n ) {
118 #else
119  for( std::size_t n = start_pos; n < MemberCount; ++n ) {
120 #endif
121  if( hashes[n] == hash ) {
122  if constexpr( do_full_name_match ) {
123  if( DAW_JSON_UNLIKELY( key != names[n].name ) ) {
124  continue;
125  }
126  }
127  return n;
128  }
129  }
130  return MemberCount;
131  }

165  /***
166  * Get the position from already seen JSON members or move the parser
167  * forward until we reach the end of the class or the member.
168  * @tparam N Number of members in json_class
169  * @tparam ParseState see IteratorRange
170  * @param locations members location and names
171  * @param parse_state Current JSON data
172  * @return IteratorRange with begin( ) being start of value
173  */
174  template<std::size_t pos, bool from_start = false, std::size_t N,
175  typename ParseState, bool B>
176  [[nodiscard]] inline constexpr ParseState
177  find_class_member( locations_info_t<N, B> &locations,
178  ParseState &parse_state, bool is_nullable,
179  daw::string_view member_name ) {
180 
181  daw_json_assert_weak( is_nullable or
182  ( not locations[pos].missing( ) ) or
183  ( not parse_state.is_closing_brace_checked( ) ),
184  missing_member( member_name ), parse_state );
185 
186  parse_state.trim_left_unchecked( );
187  while( locations[pos].missing( ) & ( parse_state.front( ) != '}' ) ) {
188  daw_json_assert_weak( parse_state.has_more( ),
189  ErrorReason::UnexpectedEndOfData, parse_state );
190  // TODO: fully unescape name
191  auto const name = parse_name( parse_state );
192  auto const name_pos =
193  locations.template find_name<( from_start ? 0 : pos )>( name );
194  if( name_pos >= std::size( locations ) ) {
195  // This is not a member we are concerned with
196  (void)skip_value( parse_state );
197  parse_state.clean_tail( );
198  continue;
199  }
200  if( name_pos == pos ) {
201  locations[pos].set_range( parse_state );
202  break;
203  } else {
204  // We are out of order, store position for later
205  // OLDTODO: use type knowledge to speed up skip
206  // OLDTODO: on skipped classes see if way to store
207  // member positions so that we don't have to
208  // re-parse them after
209  // RESULT: storing preparsed is slower, don't try 3 times
210  // it also limits the type of things we can parse potentially
211  // Using locations to switch on BaseType is slower too
212  locations[name_pos].set_range( skip_value( parse_state ) );
213 
214  parse_state.clean_tail( );
215  }
216  }
217  return locations[pos]
218  .get_range( template_arg<ParseState> )
219  .with_allocator( parse_state );
220  }
