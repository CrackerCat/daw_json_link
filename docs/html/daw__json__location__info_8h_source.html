83  template<std::size_t start_pos>
84  [[nodiscard]] constexpr std::size_t
85  find_name( daw::string_view key ) const {
86  UInt32 const hash = name_hash( key );
87 #if defined( _MSC_VER ) and not defined( __clang__ )
88  // MSVC has a bug where the list initialization isn't sequenced in
89  // order of appearance.
90  for( std::size_t n = 0; n < MemberCount; ++n ) {
91 #else
92  for( std::size_t n = start_pos; n < MemberCount; ++n ) {
93 #endif
94  if( hashes[n] == hash ) {
95  if constexpr( has_collisions ) {
96  if( DAW_JSON_UNLIKELY( key != names[n].name ) ) {
97  continue;
98  }
99  }
100  return n;
101  }
102  }
103  return MemberCount;
104  }

157  // TODO: fully unescape name

161  if( name_pos >= std::size( locations ) ) {
162  // This is not a member we are concerned with

164  parse_state.clean_tail( );

169  break;
170  } else {

172  // OLDTODO: use type knowledge to speed up skip
