115  [[nodiscard]] constexpr std::size_t
116  find_name( std::size_t start_pos, daw::string_view key ) const {
117  UInt32 const hash = name_hash( key );
118 #if defined( _MSC_VER ) and not defined( __clang__ )
119  // MSVC has a bug where the list initialization isn't sequenced in
120  // order of appearance.
121  for( std::size_t n = 0; n < MemberCount; ++n ) {
122 #else
123  for( std::size_t n = start_pos; n < MemberCount; ++n ) {
124 #endif
125  if( hashes[n] == hash ) {
126  if constexpr( do_full_name_match ) {
127  if( DAW_UNLIKELY( key != names[n].name ) ) {
128  continue;
129  }
130  }
131  return n;
132  }
133  }
134  return MemberCount;
135  }

182  /***
183  * Get the position from already seen JSON members or move the parser
184  * forward until we reach the end of the class or the member.
185  * @tparam N Number of members in json_class
186  * @tparam ParseState see IteratorRange
187  * @param locations members location and names
188  * @param parse_state Current JSON data
189  * @return IteratorRange with begin( ) being start of value
190  */
191  enum class AllMembersMustExist { yes, no };
192  template<std::size_t pos, AllMembersMustExist must_exist,
193  bool from_start = false, std::size_t N, typename ParseState,
194  bool B, typename CharT>
195  [[nodiscard]] inline constexpr ParseState
196  find_class_member( ParseState &parse_state,
197  locations_info_t<N, CharT, B> &locations,
198  bool is_nullable, daw::string_view member_name ) {
199 
200  daw_json_assert_weak( is_nullable | ( not locations[pos].missing( ) ) |
201  ( not parse_state.is_closing_brace_checked( ) ),
202  missing_member( member_name ), parse_state );
203 
204  parse_state.trim_left_unchecked( );
205  while( locations[pos].missing( ) & ( parse_state.front( ) != '}' ) ) {
206  // TODO: fully unescape name
207  // parse_name checks if we have more and are quotes
208  auto const name = parse_name( parse_state );
209  auto const name_pos =
210  locations.find_name( from_start ? 0 : pos, name );
211  if constexpr( must_exist == AllMembersMustExist::yes ) {
212  daw_json_assert_weak( name_pos < std::size( locations ),
213  ErrorReason::UnknownMember, parse_state );
214  } else {
215  if( name_pos >= std::size( locations ) ) {
216  // This is not a member we are concerned with
217  (void)skip_value( parse_state );
218  parse_state.move_next_member_or_end( );
219  continue;
220  }
221  }
222  if( name_pos == pos ) {
223  locations[pos].set_range( parse_state );
224  break;
225  } else {
226  // We are out of order, store position for later
227  // OLDTODO: use type knowledge to speed up skip
228  // OLDTODO: on skipped classes see if way to store
229  // member positions so that we don't have to
230  // re-parse them after
231  // RESULT: storing preparsed is slower, don't try 3 times
232  // it also limits the type of things we can parse potentially
233  // Using locations to switch on BaseType is slower too
234  locations[name_pos].set_range( skip_value( parse_state ) );
235 
236  if constexpr( ParseState::is_unchecked_input ) {
237  if( name_pos + 1 < std::size( locations ) ) {
238  parse_state.move_next_member( );
239  } else {
240  parse_state.move_next_member_or_end( );
241  }
242  } else {
243  parse_state.move_next_member_or_end( );
244  }
245  }
246  }
247  if constexpr( ParseState::has_allocator ) {
248  return locations[pos]
249  .get_range( template_arg<ParseState> )
250  .with_allocator( parse_state );
251  } else {
252  return locations[pos].get_range( template_arg<ParseState> );
253  }
254  }
