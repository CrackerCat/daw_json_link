115  /***
116  * Parse the current element
117  * @pre good( ) returns true
118  * @return The parsed result of ParseElement
119  */
120  [[nodiscard]] inline constexpr value_type operator*( ) const {
121  daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
122  ErrorReason::UnexpectedEndOfData, m_state );
123 
124  auto tmp = m_state;
125 
126  if constexpr( json_details::is_guaranteed_rvo_v<ParsePolicy> ) {
127  auto const ae = daw::on_exit_success( [&] { m_can_skip = tmp.first; } );
128  return json_details::parse_value<element_type>(
129  ParseTag<element_type::expected_type>{ }, tmp );
130  } else {
131  auto result = json_details::parse_value<element_type>(
132  ParseTag<element_type::expected_type>{ }, tmp );
133 
134  m_can_skip = tmp.first;
135  return result;
136  }
137  }
