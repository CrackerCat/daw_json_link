114  template<bool is_unchecked_input>
115  DAW_ATTRIBUTE_FLATTEN [[nodiscard]] static inline constexpr char const *
116  skip_digits( char const *first, char const *const last ) {
117  if( DAW_JSON_UNLIKELY( first >= last ) ) {
118  return first;
119  }
120  unsigned dig = parse_digit( *first );
121  while( dig < 10 ) {
122  ++first;
123  if constexpr( not is_unchecked_input ) {
124  if( DAW_JSON_UNLIKELY( first >= last ) ) {
125  break;
126  }
127  }
128  dig = parse_digit( *first );
129  }
130  return first;
131  }
132  /***
133  * Skip a number and store the position of it's components in the returned
134  * Range
135  */
136  template<typename Range>
137  [[nodiscard]] static constexpr Range skip_number( Range &rng ) {
138  auto result = rng;
139  char const *first = rng.first;
140  char const *const last = rng.last;
141  daw_json_assert_weak( first < last, ErrorReason::UnexpectedEndOfData, rng );
142  if( *first == '-' ) {
143  ++first;
144  }
145  first = skip_digits<Range::is_unchecked_input>( first, last );
146  char const *decimal = nullptr;
147  if( ( Range::is_unchecked_input or first < last ) and ( *first == '.' ) ) {
148  decimal = first;
149  ++first;
150  first = skip_digits<Range::is_unchecked_input>( first, last );
151  }
152  char const *exp = nullptr;
153  unsigned dig = parse_digit( *first );
154  if( ( Range::is_unchecked_input or ( first < last ) ) &
155  ( ( dig == parsed_constants::e_char ) |
156  ( dig == parsed_constants::E_char ) ) ) {
157  exp = first;
158  ++first;
159  daw_json_assert_weak( first < last, ErrorReason::UnexpectedEndOfData,
160  [&] {
161  auto r = rng;
162  r.first = first;
163  return r;
164  }( ) );
165  dig = parse_digit( *first );
166  if( ( dig == parsed_constants::plus_char ) |
167  ( dig == parsed_constants::minus_char ) ) {
168  ++first;
169  }
170  first = skip_digits<Range::is_unchecked_input>( first, last );
171  }
172 
173  rng.first = first;
174  result.last = first;
175  result.class_first = decimal;
176  result.class_last = exp;
177  return result;
178  }

180  /***
181  * When we don't know ahead of time what we are skipping switch on the first
182  * value and call that types specific skipper
183  * TODO: Investigate if there is a difference for the times we know what the
184  * member should be if that can increase performance
185  */
186  template<typename Range>
187  [[nodiscard]] static inline constexpr Range skip_value( Range &rng ) {
188  daw_json_assert_weak( rng.has_more( ), ErrorReason::UnexpectedEndOfData,
189  rng );
190 
191  // reset counter
192  rng.counter = 0;
193  switch( rng.front( ) ) {

195  return skip_string( rng );
196  case '[':
197  return rng.skip_array( );
198  case '{':
199  return rng.skip_class( );
200  case 't':
201  return skip_true( rng );
202  case 'f':
203  return skip_false( rng );
204  case 'n':
205  return skip_null( rng );
206  case '-':
207  case '0':
208  case '1':
209  case '2':
210  case '3':
211  case '4':
212  case '5':
213  case '6':
214  case '7':
215  case '8':
216  case '9':
217  return skip_number( rng );
218  }
219  if constexpr( Range::is_unchecked_input ) {
220  DAW_UNREACHABLE( );
221  } else {
222  daw_json_error( ErrorReason::InvalidStartOfValue, rng );
