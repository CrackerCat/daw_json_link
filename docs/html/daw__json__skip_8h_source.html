122  template<bool skip_end_check>
123  DAW_ATTRIBUTE_FLATTEN [[nodiscard]] inline constexpr char const *
124  skip_digits( char const *first, char const *const last ) {
125  unsigned dig = parse_digit( *first );
126  while( dig < 10 ) {
127  ++first;
128  if constexpr( not skip_end_check ) {
129  if( DAW_JSON_UNLIKELY( first >= last ) ) {
130  break;
131  }
132  }
133  dig = parse_digit( *first );
134  }
135  return first;
136  }
137  /***
138  * Skip a number and store the position of it's components in the returned
139  * ParseState
140  */
141  template<typename ParseState>
142  [[nodiscard]] constexpr ParseState
143  skip_number( ParseState &parse_state ) {
144  daw_json_assert_weak( parse_state.has_more( ),
145  ErrorReason::UnexpectedEndOfData, parse_state );
146 
147  auto result = parse_state;
148  char const *first = parse_state.first;
149  char const *const last = parse_state.last;
150  if( *first == '-' ) {
151  ++first;
152  }
153  if( DAW_JSON_LIKELY( first < last ) ) {
154  first =
155  skip_digits<( ParseState::is_zero_terminated_string or
156  ParseState::is_unchecked_input )>( first, last );
157  }
158  char const *decimal = nullptr;
159  if( ( ( ParseState::is_zero_terminated_string or
160  ParseState::is_unchecked_input ) or
161  first < last ) and
162  ( *first == '.' ) ) {
163  decimal = first;
164  ++first;
165  if( DAW_JSON_LIKELY( first < last ) ) {
166  first =
167  skip_digits<( ParseState::is_zero_terminated_string or
168  ParseState::is_unchecked_input )>( first, last );
169  }
170  }
171  char const *exp = nullptr;
172  unsigned dig = parse_digit( *first );
173  if( ( ( ParseState::is_zero_terminated_string or
174  ParseState::is_unchecked_input ) or
175  ( first < last ) ) &
176  ( ( dig == parsed_constants::e_char ) |
177  ( dig == parsed_constants::E_char ) ) ) {
178  exp = first;
179  ++first;
180  daw_json_assert_weak( first < last, ErrorReason::UnexpectedEndOfData,
181  [&] {
182  auto r = parse_state;
183  r.first = first;
184  return r;
185  }( ) );
186  dig = parse_digit( *first );
187  if( ( dig == parsed_constants::plus_char ) |
188  ( dig == parsed_constants::minus_char ) ) {
189  ++first;
190  }
191  if( DAW_JSON_LIKELY( first < last ) ) {
192  first =
193  skip_digits<( ParseState::is_zero_terminated_string or
194  ParseState::is_unchecked_input )>( first, last );
195  }
196  }
197 
198  parse_state.first = first;
199  result.last = first;
200  result.class_first = decimal;
201  result.class_last = exp;
202  return result;
203  }

205  /***
206  * When we don't know ahead of time what we are skipping switch on the
207  * first value and call that types specific skipper
208  * TODO: Investigate if there is a difference for the times we know what
209  * the member should be if that can increase performance
210  */
211  template<typename ParseState>
212  [[nodiscard]] inline constexpr ParseState
213  skip_value( ParseState &parse_state ) {
214  daw_json_assert_weak( parse_state.has_more( ),
215  ErrorReason::UnexpectedEndOfData, parse_state );
216 
217  // reset counter
218  parse_state.counter = 0;
219  switch( parse_state.front( ) ) {
220  case '"':
221  return skip_string( parse_state );
222  case '[':
223  return parse_state.skip_array( );
224  case '{':
225  return parse_state.skip_class( );
226  case 't':
227  return skip_true( parse_state );
228  case 'f':
229  return skip_false( parse_state );
230  case 'n':
231  return skip_null( parse_state );
232  case '-':
233  case '0':
234  case '1':
235  case '2':
236  case '3':
237  case '4':
238  case '5':
239  case '6':
240  case '7':
241  case '8':
242  case '9':
243  return skip_number( parse_state );
244  case '\0':
245  daw_json_error( ErrorReason::InvalidStartOfValue, parse_state );
246  }
247  if constexpr( ParseState::is_unchecked_input ) {
248  DAW_UNREACHABLE( );
249  } else {
250  daw_json_error( ErrorReason::InvalidStartOfValue, parse_state );
251  }
252  }

254  /***
255  * Used in json_array_iterator::operator++( ) as we know the type we are
256  * skipping
257  */
258  template<typename JsonMember, typename ParseState>
259  [[nodiscard]] DAW_ATTRIBUTE_FLATTEN inline constexpr ParseState
260  skip_known_value( ParseState &parse_state ) {
261  daw_json_assert_weak( parse_state.has_more( ),
262  ErrorReason::UnexpectedEndOfData, parse_state );
263  if constexpr( JsonMember::expected_type == JsonParseTypes::Date or
264  JsonMember::expected_type == JsonParseTypes::StringRaw or
265  JsonMember::expected_type ==
266  JsonParseTypes::StringEscaped or
267  JsonMember::expected_type == JsonParseTypes::Custom ) {
268  // json string encodings
269  daw_json_assert_weak( parse_state.front( ) == '"',
270  ErrorReason::InvalidString, parse_state );
271  parse_state.remove_prefix( );
272  return json_details::skip_string_nq( parse_state );
273  } else if constexpr( JsonMember::expected_type ==
274  JsonParseTypes::Real or
275  JsonMember::expected_type ==
276  JsonParseTypes::Signed or
277  JsonMember::expected_type ==
278  JsonParseTypes::Unsigned or
279  JsonMember::expected_type ==
280  JsonParseTypes::Bool or
281  JsonMember::expected_type ==
282  JsonParseTypes::Null ) {
283  // All literals
284  return skip_number( parse_state );
285  } else if constexpr( JsonMember::expected_type ==
286  JsonParseTypes::Array ) {
287  daw_json_assert_weak( parse_state.is_opening_bracket_checked( ),
288  ErrorReason::InvalidArrayStart, parse_state );
289  return parse_state.skip_array( );
290  } else if constexpr( JsonMember::expected_type ==
291  JsonParseTypes::Class ) {
292  daw_json_assert_weak( parse_state.is_opening_brace_checked( ),
293  ErrorReason::InvalidClassStart, parse_state );
294  return parse_state.skip_class( );
295  } else {
296  // Woah there
297  static_assert( JsonMember::expected_type == JsonParseTypes::Class,
298  "Unknown JsonParseTypes value. This is a programmer "
299  "error and the preceding did not check for it" );
300  DAW_UNREACHABLE( );
301  }
