56  // https://stackoverflow.com/questions/28868367/getting-the-high-part-of-64-bit-integer-multiplication
57  DAW_ATTRIBUTE_FLATTEN inline constexpr std::uint64_t
58  Emulate64x64to128( std::uint64_t &r_hi, std::uint64_t const x,
59  std::uint64_t const y ) {
60  std::uint64_t const x0 = static_cast<std::uint32_t>( x );
61  std::uint64_t const x1 = x >> 32;
62  std::uint64_t const y0 = static_cast<std::uint32_t>( y );
63  std::uint64_t const y1 = y >> 32;
64  std::uint64_t const p11 = x1 * y1;
65  std::uint64_t const p01 = x0 * y1;
66  std::uint64_t const p10 = x1 * y0;
67  std::uint64_t const p00 = x0 * y0;
68 
69  // 64-bit product + two 32-bit values
70  std::uint64_t const middle =
71  p10 + ( p00 >> 32 ) + static_cast<std::uint32_t>( p01 );
72 
73  // 64-bit product + two 32-bit values
74  r_hi = p11 + ( middle >> 32 ) + ( p01 >> 32 );
75 
76  // Add LOW PART and lower half of MIDDLE PART
77  return ( middle << 32 ) | static_cast<std::uint32_t>( p00 );
78  }

1026  // Attempts to compute i * 10^(power) exactly; and if "negative" is
1027  // true, negate the result.
1028  // This function will only work in some cases, when it does not work,
1029  // success is set to false. This should work *most of the time* (like
1030  // 99% of the time). We assume that power is in the
1031  // [FASTFLOAT_SMALLEST_POWER, FASTFLOAT_LARGEST_POWER] interval: the
1032  // caller is responsible for this check.
1033  template<typename ExecTag>
1034  constexpr double compute_float_64( ExecTag exec_tag, std::int64_t power,
1035  std::uint64_t whole, double sign ) {
1036 
1037  if( ( power < daw::numeric_limits<double>::min_exponent10 ) |
1038  ( power > daw::numeric_limits<double>::max_exponent10 ) ) {
1039  return daw::cxmath::copy_sign(
1040  daw::json::json_details::power10<double>(
1041  exec_tag, static_cast<double>( whole ), power ),
1042  sign );
1043  }
1044  // we start with a fast path
1045  // It was described in
1046  // Clinger WD. How to read floating point numbers accurately.
1047  // ACM SIGPLAN Notices. 1990
1048 #if( FLT_EVAL_METHOD != 1 ) and ( FLT_EVAL_METHOD != 0 )
1049  // we do not trust the divisor
1050  if( ( 0 <= power ) & ( power <= 22 ) & ( whole <= 9007199254740991 ) ) {
1051 #else
1052  if( ( ( -22 <= power ) & ( power <= 22 ) ) &
1053  ( whole <= 9'007'199'254'740'991 ) ) {
1054 #endif
1055  // convert the integer into a double. This is lossless since
1056  // 0 <= i <= 2^53 - 1.
1057  // The general idea is as follows.
1058  // If 0 <= s < 2^53 and if 10^0 <= p <= 10^22 then
1059  // 1) Both s and p can be represented exactly as 64-bit
1060  // floating-point values (binary64). 2) Because s and p can be
1061  // represented exactly as floating-point values, then s * p and s /
1062  // p will produce correctly rounded values.
1063  //
1064  return daw::cxmath::copy_sign(
1065  daw::json::json_details::power10<double>(
1066  exec_tag, static_cast<double>( whole ), power ),
1067  sign );
1068  }
1069  // When 22 < power && power < 22 + 16, we could
1070  // hope for another, secondary fast path. It wa
1071  // described by David M. Gay in "Correctly rounded
1072  // binary-decimal and decimal-binary conversions." (1990)
1073  // If you need to compute i * 10^(22 + x) for x < 16,
1074  // first compute i * 10^x, if you know that result is exact
1075  // (e.g., when i * 10^x < 2^53),
1076  // then you can still proceed and do (i * 10^x) * 10^22.
1077  // Is this worth your time?
1078  // You need 22 < power *and* power < 22 + 16 *and* (i * 10^(x-22) <
1079  // 2^53) for this second fast path to work. If you you have 22 < power
1080  // *and* power < 22 + 16, and then you optimistically compute "i *
1081  // 10^(x-22)", there is still a chance that you have wasted your time
1082  // if i * 10^(x-22) >= 2^53. It makes the use cases of this
1083  // optimization maybe less common than we would like. Source:
1084  // http://www.exploringbinary.com/fast-path-decimal-to-floating-point-conversion/
1085  // also used in RapidJSON: https://rapidjson.org/strtod_8h_source.html
1086 
1087  // The fast path has now failed, so we are failing back on the slower
1088  // path.
1089 
1090  // In the slow path, we need to adjust i so that it is > 1<<63 which
1091  // is always possible, except if i == 0, so we handle i == 0
1092  // separately.
1093  if( whole == 0 ) {
1094  return daw::cxmath::copy_sign( 0.0, sign );
1095  }
1096  // We are going to need to do some 64-bit arithmetic to get a more
1097  // precise product. We use a table lookup approach. It is safe because
1098  // power >= FASTFLOAT_SMALLEST_POWER and power <=
1099  // FASTFLOAT_LARGEST_POWER We recover the mantissa of the power, it
1100  // has a leading 1. It is always rounded down.
1101 
1102  std::uint64_t factor_mantissa = [&] {
1103  auto const idx = power - FASTFLOAT_SMALLEST_POWER;
1104  return math_const::mantissa_64[idx];
1105  }( );
1106 
1107  // The exponent is 1024 + 63 + power
1108  // + floor(log(5**power)/log(2)).
1109  // The 1024 comes from the ieee64 standard.
1110  // The 63 comes from the fact that we use a 64-bit word.
1111  //
1112  // Computing floor(log(5**power)/log(2)) could be
1113  // slow. Instead we use a fast function.
1114  //
1115  // For power in (-400,350), we have that
1116  // (((152170 + 65536) * power ) >> 16);
1117  // is equal to
1118  // floor(log(5**power)/log(2)) + power when power >= 0
1119  // and it is equal to
1120  // ceil(log(5**-power)/log(2)) + power when power < 0
1121  //
1122  //
1123  // The 65536 is (1<<16) and corresponds to
1124  // (65536 * power) >> 16 ---> power
1125  //
1126  // ((152170 * power ) >> 16) is equal to
1127  // floor(log(5**power)/log(2))
1128  //
1129  // Note that this is not magic: 152170/(1<<16) is
1130  // approximately equal to log(5)/log(2).
1131  // The 1<<16 value is a power of two; we could use a
1132  // larger power of 2 if we wanted to.
1133  //
1134  std::int64_t const exponent =
1135  ( ( ( 152'170 + 65'536 ) * power ) >> 16 ) + 1024 + 63;
1136  // We want the most significant bit of i to be 1. Shift if needed.
1137  int lz = leading_zeroes( exec_tag, whole );
1138  auto const whole_orig = whole;
1139  whole <<= lz;
1140  // We want the most significant 64 bits of the product. We know
1141  // this will be non-zero because the most significant bit of i is
1142  // 1.
1143 
1144  auto [lower, upper] =
1145  full_multiplication( exec_tag, whole, factor_mantissa );
1146  // We know that upper has at most one leading zero because
1147  // both i and factor_mantissa have a leading one. This means
1148  // that the result is at least as large as ((1<<63)*(1<<63))/(1<<64).
1149 
1150  // As long as the first 9 bits of "upper" are not "1", then we
1151  // know that we have an exact computed value for the leading
1152  // 55 bits because any imprecision would play out as a +1, in
1153  // the worst case.
1154  // Having 55 bits is necessary because
1155  // we need 53 bits for the mantissa but we have to have one rounding
1156  // bit and we can waste a bit if the most significant bit of the
1157  // product is zero. We expect this next branch to be rarely taken (say
1158  // 1% of the time). When (upper & 0x1FF) == 0x1FF, it can be common
1159  // for lower + i < lower to be true (proba. much higher than 1%).
1160  if( DAW_JSON_UNLIKELY( ( upper & 0x1FF ) == 0x1FF ) &
1161  ( lower + whole < lower ) ) {
1162  std::uint64_t factor_mantissa_low =
1163  math_const::mantissa_128[power - FASTFLOAT_SMALLEST_POWER];
1164  // next, we compute the 64-bit x 128-bit multiplication, getting a
1165  // 192-bit result (three 64-bit values)
1166 
1167  auto const [product_low, product_middle2] =
1168  full_multiplication( exec_tag, whole, factor_mantissa_low );
1169  std::uint64_t const product_middle1 = lower;
1170  std::uint64_t product_high = upper;
1171  std::uint64_t const product_middle =
1172  product_middle1 + product_middle2;
1173 
1174  if( product_middle < product_middle1 ) {
1175  product_high++; // overflow carry
1176  }
1177  // we want to check whether mantissa *i + i would affect our result
1178  // This does happen, e.g. with 7.3177701707893310e+15
1179  if( ( ( product_middle + 1 == 0 ) &&
1180  ( ( product_high & 0x1FF ) == 0x1FF ) &&
1181  ( product_low + whole <
1182  product_low ) ) ) { // let us be prudent and bail out.
1183  return daw::cxmath::copy_sign(
1184  daw::json::json_details::power10<double>(
1185  exec_tag, static_cast<double>( whole_orig ), power ),
1186  sign );
1187  }
1188  upper = product_high;
1189  lower = product_middle;
1190  }
1191  // The final mantissa should be 53 bits with a leading 1.
1192  // We shift it so that it occupies 54 bits with a leading 1.
1194  std::uint64_t const upperbit = upper >> 63;
1195  std::uint64_t mantissa = upper >> ( upperbit + 9 );
1196  lz += static_cast<int>( 1 ^ upperbit );
1197  // Here we have mantissa < (1<<54).
1198 
1199  // We have to round to even. The "to even" part
1200  // is only a problem when we are right in between two floats
1201  // which we guard against.
1202  // If we have lots of trailing zeros, we may fall right between two
1203  // floating-point values.
1204 
1205  // if mantissa & 1 == 1 we might need to round up.
1206  //
1207  // Scenarios:
1208  // 1. We are not in the middle. Then we should round up.
1209  //
1210  // 2. We are right in the middle. Whether we round up depends
1211  // on the last significant bit: if it is "one" then we round
1212  // up (round to even) otherwise, we do not.
1213  //
1214  // So if the last significant bit is 1, we can safely round up.
1215  // Hence we only need to bail out if (mantissa & 3) == 1.
1216  // Otherwise we may need more accuracy or analysis to determine
1217  // whether we are exactly between two floating-point numbers. It can
1218  // be triggered with 1e23. Note: because the factor_mantissa and
1219  // factor_mantissa_low are almost always rounded down (except for
1220  // small positive powers), almost always should round up.
1221  if( DAW_JSON_UNLIKELY( ( lower == 0 ) & ( ( upper & 0x1FF ) == 0 ) &
1222  ( ( mantissa & 3 ) == 1 ) ) ) {
1223  return daw::cxmath::copy_sign(
1224  daw::json::json_details::power10<double>(
1225  exec_tag, static_cast<double>( whole_orig ), power ),
1226  sign );
1227  }
1228 
1229  mantissa += mantissa & 1;
1230  mantissa >>= 1;
1231  // Here we have mantissa < (1<<53), unless there was an overflow
1232  if( mantissa >= ( 1ULL << 53 ) ) {
1234  // This will happen when parsing values such
1235  // as 7.2057594037927933e+16
1237  mantissa = ( 1ULL << 52 );
