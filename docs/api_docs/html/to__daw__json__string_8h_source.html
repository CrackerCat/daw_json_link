124  /***
125  * This is the default ToJsonConverter for json_raw. By default is will
126  * return the stringified version of the value if, to_string( T ) exists.
127  * Otherwise it will fallback to an std::ostream converter for T if it
128  * exists.
129  * @tparam T type of value to convert to a string
130  */
131  template<typename T>
132  struct default_to_json_converter_t {
133  template<typename U>
134  [[nodiscard]] static inline auto use_stream( U const &v ) {
135  std::stringstream ss{ };
136  ss << v;
137  return DAW_MOVE( ss ).str( );
138  }

140  template<typename U>
141  [[nodiscard]] inline constexpr auto operator( )( U const &value ) const {
142  if constexpr( json_details::is_string_view_like_v<U> ) {
143  return std::string_view( std::data( value ), std::size( value ) );
144  } else if constexpr( json_details::to_strings::has_to_string_v<U> ) {
145  using json_details::to_strings::to_string;
146  return to_string( value );
147  } else if constexpr( json_details::has_ostream_op_v<U> ) {
148  return use_stream( value );
149  } else if constexpr( std::is_convertible_v<U, std::string_view> ) {
150  return static_cast<std::string_view>( value );
151  } else if constexpr( std::is_convertible_v<U, std::string> ) {
152  return static_cast<std::string>( value );
153  } else if constexpr( daw::is_arithmetic_v<U> ) {
154  return to_string( value );
155  } else if constexpr( std::is_enum_v<U> ) {
156  return to_string( static_cast<std::underlying_type_t<U>>( value ) );
157  } else if constexpr( json_details::can_deref_v<U> ) {
158  static_assert( json_details::has_op_bool_v<U>,
159  "default_to_converter cannot work with type" );
160  using deref_t = daw::remove_cvref_t<decltype( *value )>;
161  if constexpr( json_details::is_string_view_like_v<deref_t> ) {
162  if( value ) {
163  auto const &v = *value;
164  return std::string_view( std::data( v ), std::size( v ) );
165  }
166  return std::string_view( "null", 4 );
167  } else if constexpr( json_details::to_strings::has_to_string_v<
168  deref_t> ) {
169  if( value ) {
170  using json_details::to_strings::to_string;
171  return to_string( *value );
172  } else {
173  using result_t =
174  daw::remove_cvref_t<decltype( to_string( *value ) )>;
175  return result_t{ "null" };
176  }
177  } else if constexpr( std::is_convertible_v<deref_t,
178  std::string_view> ) {
179  if( value ) {
180  return static_cast<std::string_view>( value );
181  }
182  return std::string_view{ "null" };
183  } else if constexpr( std::is_convertible_v<deref_t, std::string> ) {
184  if( value ) {
185  return static_cast<std::string>( value );
186  }
187  return std::string( "null" );
188  } else {
189  if( value ) {
190  return use_stream( *value );
191  } else {
192  return std::string( "null" );
193  }
194  }
195  }
196  }

354  template<typename OutputIterator>
355  constexpr void utf32_to_utf8( std::uint32_t cp, OutputIterator &it ) {
356  if( cp <= 0x7FU ) {
357  *it++ = static_cast<char>( cp );
358  return;
359  }
360  if( cp <= 0x7FFU ) {
361  *it++ = static_cast<char>( ( cp >> 6U ) | 0b11000000U );
362  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
363  return;
364  }
365  if( cp <= 0xFFFFU ) {
366  *it++ = static_cast<char>( ( cp >> 12U ) | 0b11100000U );
367  *it++ =
368  static_cast<char>( ( ( cp >> 6U ) & 0b00111111U ) | 0b10000000U );
369  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
370  return;
371  }

373  *it++ = static_cast<char>( ( cp >> 18U ) | 0b11110000U );
